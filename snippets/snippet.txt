[image convert raw to png] find . -name "*\%#.raw" | parallel --no-notice -a - convert -depth 8 -size 1024x1024+0 gray:{} {}.png
[image convert png to raw] convert white.png -depth 8 -size 1000x1000 gray:white.raw
